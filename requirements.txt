# LiteCAD Dependencies

# GUI Framework
PySide6>=6.5.0

# 3D Visualization (PyVista - empfohlen)
pyvista>=0.43.0
pyvistaqt>=0.11.0

# 3D CAD Kernel (build123d - empfohlen)
build123d>=0.7.0

# Geometrie-Operationen
shapely>=2.0.0

# Numerik
numpy>=1.24.0

# OpenGL (nur für Fallback ohne PyVista)
PyOpenGL>=3.1.6

# ================================================
# Installation:
# ================================================
# pip install -r requirements.txt
#
# Oder nur die essentiellen Pakete:
# pip install PySide6 pyvista pyvistaqt build123d shapely numpy
#
# ================================================
# Warum diese Bibliotheken?
# ================================================
#
# PyVista + pyvistaqt:
#   - Hardware-beschleunigtes 3D-Rendering (VTK)
#   - Eingebautes Face-Picking
#   - ViewCube/Orientation Widget
#   - Kein Z-Fighting
#   - Skaliert auf Millionen Faces
#
# build123d:
#   - Echtes parametrisches CAD-Modeling
#   - Boolean-Operationen (Join, Cut, Intersect)
#   - Fillets, Chamfers
#   - STEP/STL Export
#   - Basiert auf OpenCascade (Industriestandard)
#
# Shapely:
#   - 2D-Geometrie-Operationen
#   - Polygon-Erkennung aus Linien
#   - Face-Detection für Extrusion
